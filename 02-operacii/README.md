# Операции

## Строковые

    $name = "Иван";
    $sub_name = "Петров";
    echo "$name $sub_name";       // Иван Петров
    echo $name . ' ' . $sub_name; // Иван Петров (используем конкатенацию)

`.` - конкатенация, склеивание строк.

## Математические

    $a = 3;
    $b = 5;
    echo 3 + 5;   // 8
    echo $a + $b; // 8

## Сравнения

    == - проверка на равенство
    === - проверка на строгое равенство
    >
    >=
    <=

## Логические

    && - и
    || - или
    !  - не
